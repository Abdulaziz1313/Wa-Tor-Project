13// Params holds all configuration parameters for a single run of the
14// Wa-Tor simulation. These values are set from command–line flags.
15type Params struct {
16 NumShark int // starting population of sharks
17 NumFish int // starting population of fish
18 FishBreed int // chronons before a fish can reproduce
19 SharkBreed int // chronons before a shark can reproduce
20 Starve int // chronons a shark can survive without food
21 GridSize int // width and height of the toroidal grid (GridSize x GridSize)
22 Threads int // number of goroutines to use for the parallel step
23 Steps int // number of simulation steps (chronons) to run
24 PrintEvery int // how often to print the world in text mode (0 = never)

30// parseParams parses command–line flags into a Params struct and performs
31// basic validation of the input values.
32func parseParams() Params {
33 p := Params{}
34
35 flag.IntVar(&p.NumShark, "numShark", 100, "Starting population of sharks")
36 flag.IntVar(&p.NumFish, "numFish", 200, "Starting population of fish")
37 flag.IntVar(&p.FishBreed, "fishBreed", 3, "Chronons before a fish can reproduce")
38 flag.IntVar(&p.SharkBreed, "sharkBreed", 5, "Chronons before a shark can reproduce")
39 flag.IntVar(&p.Starve, "starve", 3, "Chronons a shark can live without food")
40 flag.IntVar(&p.GridSize, "gridSize", 20, "Grid dimension (NxN)")
41 flag.IntVar(&p.Threads, "threads", 1, "Number of threads (goroutines) to use")
42 flag.IntVar(&p.Steps, "steps", 200, "Number of simulation steps (chronons)")
43 flag.IntVar(&p.PrintEvery, "printEvery", 20, "How often to print the grid (0 = never)")
44 flag.StringVar(&p.CSVFile, "csv", "", "Optional CSV file to write stats (e.g. stats.csv)")
45 flag.BoolVar(&p.Graphics, "graphics", false, "Run with graphical window (Ebiten)")
46
47 flag.Parse()
48
49 if p.NumShark < 0 || p.NumFish < 0 || p.GridSize <= 0 {
50 fmt.Println("Error: invalid parameter values")
51 os.Exit(1)
52 }
53 if p.Steps <= 0 {
54 fmt.Println("Error: steps must be > 0")
55 os.Exit(1)
56 }
57
58 return p
59}

91// RunSimulation executes the Wa-Tor simulation in text mode.
92// It optionally writes population statistics to a CSV file.
93func RunSimulation(p Params) {
94 world := NewWorld(p)
95
96 var csvWriter *bufio.Writer
97 var csvFile *os.File
98 var err error
99
100 if p.CSVFile != "" {
101 csvFile, err = os.Create(p.CSVFile)
102 if err != nil {
103 fmt.Println("Error creating CSV file:", err)
104 os.Exit(1)
105 }
106 defer csvFile.Close()
107
108 csvWriter = bufio.NewWriter(csvFile)
109 defer csvWriter.Flush()
110
111 // CSV header: step, fish count, shark count
112 fmt.Fprintln(csvWriter, "step,fish,sharks")
113 }
114
115 start := time.Now()
116
117 for step := 0; step < p.Steps; step++ {
118 fish, sharks := world.Count()
119
120 // Log stats to CSV if requested.
121 if csvWriter != nil {
122 fmt.Fprintf(csvWriter, "%d,%d,%d\n", step, fish, sharks)
123 }
124
125 // Optionally print the world in ASCII.
126 if p.PrintEvery > 0 && step%p.PrintEvery == 0 {
127 clearScreen()
128 fmt.Printf("Step %d\n", step)
129 fmt.Printf("Fish=%d Sharks=%d\n", fish, sharks)
130 world.PrintColored()
131 time.Sleep(50 * time.Millisecond) // small delay so animation is visible
132 }
133
134 // Sequential vs parallel step.
135 if p.Threads > 1 {
136 world.StepParallel(p.Threads)
137 } else {
138 world.Step()
139 }
140 }
141
142 elapsed := time.Since(start)
143 fmt.Printf("\nSimulation finished in %v\n", elapsed)
144}

146// clearScreen clears the terminal using the appropriate mechanism for the
147// current operating system. On Windows it calls "cls"; on Unix-like systems
148// it writes ANSI escape codes.
149func clearScreen() {

157 }
